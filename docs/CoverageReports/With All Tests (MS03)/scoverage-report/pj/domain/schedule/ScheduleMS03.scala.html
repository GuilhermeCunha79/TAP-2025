109   /**
110    * Recursively schedules all tasks until no more tasks are ready or schedulable
111    */
112   def scheduleAllTasks(
113     state: SchedulingState,
114     allTasks: List[TaskInfo],
115     physicalResources: List[PhysicalResource],
116     humanResources: List[HumanResource]
117   ): Result[SchedulingState] =
118     state.readyTasks match
119       case Nil => Right(state)
120       case _ =>
121         for {
122           nextState <- scheduleNextBatch(state, physicalResources, humanResources)
123           updatedTasks = updateReadyTasks(nextState, allTasks)
124           finalState <- updatedTasks.schedules.lengthIs > state.schedules.length match
125             case true => scheduleAllTasks(updatedTasks, allTasks, physicalResources, humanResources)
126             case false => advanceTimeAndRetry(state, physicalResources, humanResources, allTasks)
127         } yield finalState

129   /**
130    * Advances time when no tasks can be scheduled at current time and retries scheduling
131    */
132   def advanceTimeAndRetry(
133     state: SchedulingState,
134     physicalResources: List[PhysicalResource],
135     humanResources: List[HumanResource],
136     allTasks: List[TaskInfo]
137   ): Result[SchedulingState] =
138     val minEarliestStart = state.readyTasks.map(_.earliestStart.to).minOption.getOrElse(0)
139     val nextResourceTime = state.resourceAvailability.values.filter(_ > minEarliestStart).minOption
140 
141     nextResourceTime.fold[Result[SchedulingState]](
142       Left(DomainError.ImpossibleSchedule)
143     ) { nextTime =>
144       val advancedTasks = state.readyTasks.map { t =>
145         val newStart = EarliestStartTime.from(nextTime).getOrElse(t.earliestStart)
146         t.copy(earliestStart = newStart)
147       }
148       val advancedState = state.copy(readyTasks = advancedTasks)
149       scheduleAllTasks(advancedState, allTasks, physicalResources, humanResources)
150     }

223    * Recursively schedules a batch of tasks, attempting each one sequentially
224    */
225   def scheduleBatchRecursively(
226     state: SchedulingState,
227     tasks: List[TaskInfo],
228     currentTime: Int,
229     availablePhysical: List[PhysicalResource],
230     availableHuman: List[HumanResource],
231     usedPhysicalIds: Set[PhysicalResourceId],
232     usedHumanIds: Set[HumanResourceId]
233   ): Result[SchedulingState] =
234     tasks match
235       case Nil => Right(state)
236       case taskInfo :: remainingTasks =>
237         tryScheduleTask(taskInfo, currentTime, availablePhysical, availableHuman, usedPhysicalIds, usedHumanIds)
238           .fold(
239             _ => scheduleBatchRecursively(
240               state, remainingTasks, currentTime, availablePhysical, availableHuman,
241               usedPhysicalIds, usedHumanIds
242             ),
243             { case (physicalIds, humanIds) =>
244               createTaskSchedule(taskInfo, currentTime, physicalIds, humanIds).flatMap { schedule =>
245                 val newState = updateStateAfterScheduling(state, taskInfo, schedule, currentTime)
246                 val updatedUsedPhysical = usedPhysicalIds ++ physicalIds.toSet
247                 val updatedUsedHuman = usedHumanIds ++ humanIds.toSet
248                 val filteredRemaining = remainingTasks.filterNot(isMatchingTask(_, taskInfo))
249 
250                 scheduleBatchRecursively(
251                   newState, filteredRemaining, currentTime, availablePhysical, availableHuman,
252                   updatedUsedPhysical, updatedUsedHuman

283   private def allocateResources[T, R](

286     usedIds: Set[R],

294   /**
295    * Generic recursive loop for sequential resource allocation
296    */
297   @tailrec
298   private def allocateResourcesLoop[T, R](
299     remaining: List[PhysicalResourceType],
300     availableResources: List[T],
301     allocated: List[R],
302     used: Set[R],
303     extractId: T => R,
304     matchesType: (T, PhysicalResourceType) => Boolean
305   ): Result[List[R]] = remaining match
306     case Nil => Right(allocated.reverse)
307 
308     case requiredType :: rest =>
309       availableResources.find(res =>
310         matchesType(res, requiredType) && !used.contains(extractId(res))
311       ) match
312         case Some(resource) =>
313           val id = extractId(resource)
314           allocateResourcesLoop(rest, availableResources, id :: allocated, used + id, extractId, matchesType)
315         case None =>
316           Left(DomainError.ImpossibleSchedule)

444       schedules = schedule :: state.schedules,
