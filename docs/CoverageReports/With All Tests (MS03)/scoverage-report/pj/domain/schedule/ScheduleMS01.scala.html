11   def allocatePhysicalResources(
12        taskId: TaskId,
13        requiredTypes: List[PhysicalResourceType],
14        availableResources: List[PhysicalResource]
15      ): Result[List[PhysicalResourceId]] =
16     requiredTypes.foldLeft[Result[(List[PhysicalResourceId], Set[PhysicalResourceId])]](Right((Nil, Set.empty))) {
17       case (accResult, requiredType) => for {
18         (assignedIds, usedIds) <- accResult
19         resource <- availableResources
20           .find(res => res.physical_type == requiredType && !usedIds.contains(res.id))
21           .toRight(DomainError.ResourceUnavailable(taskId.to, requiredType.to))
22       } yield (resource.id :: assignedIds, usedIds + resource.id)
