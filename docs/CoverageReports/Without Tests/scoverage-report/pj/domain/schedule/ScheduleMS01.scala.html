39   private def allocatePhysicalResources(
40     taskId: TaskId,
41     requiredTypes: List[String],
42     availableResources: List[PhysicalResource]
43   ): Result[List[PhysicalResourceId]] =
44     requiredTypes.foldLeft[Result[(List[PhysicalResourceId], Set[PhysicalResourceId])]](Right((Nil, Set.empty))) {
45       case (accResult, requiredType) => accResult.flatMap { case (assignedIds, usedIds) =>
46         availableResources
47           .find(resource => resource.name.equals(requiredType) && !usedIds.contains(resource.id))
48           .map(_.id)
49           .toRight(DomainError.ResourceUnavailable(taskId.to, requiredType))
50           .map(id => (id :: assignedIds, usedIds + id))
51       }
52     }.map(_._1.reverse)

54   private def allocateHumanResources(
55     taskId: TaskId,
56     requiredTypes: List[String],
57     availableHumans: List[HumanResource]
58   ): Result[List[String]] =
59     requiredTypes.foldLeft[Result[(List[String], Set[String])]](Right((Nil, Set.empty))) {
60       case (accResult, requiredType) => accResult.flatMap { case (assignedNames, usedNames) =>
61         availableHumans
62           .find(hr => hr.physicalResources.contains(requiredType) && !usedNames.contains(hr.name))
63           .toRight(DomainError.ResourceUnavailable(taskId.to, requiredType))
64           .map(hr => (hr.name :: assignedNames, usedNames + hr.name))
65       }
66     }.map(_._1.reverse)

68   private def generateSchedule(
69       physicalResources: List[PhysicalResource],
70       physicalTypes: List[String],
71       allTasks: List[Task],
72       humanResources: List[HumanResource],
73       allProducts: List[Product],
74       allOrders: List[Order]
75     ): Result[Elem] =
76 
77     type Accumulator = (List[TaskSchedule], Int)
78     val initialAccumulator: Result[Accumulator] = Right((Nil, 0))
79 
80     val scheduleResult = allOrders.foldLeft[Result[Accumulator]](initialAccumulator) { (accumulatorForOrders, currentOrder) =>
81       accumulatorForOrders.flatMap { case (scheduledTasksSoFar, globalTimeSoFar) =>
82         val matchedProduct = allProducts.find(_.id == currentOrder.productId)
83           .toRight(DomainError.ProductDoesNotExist(currentOrder.productId.to))
84 
85         (1 to currentOrder.quantity.to).foldLeft[Result[Accumulator]](Right((scheduledTasksSoFar, globalTimeSoFar))):
86           case (accumulatorForProducts, productInstanceNumber) => accumulatorForProducts.flatMap:
87             case (tasksForProduct, timeAtStartOfProduct) => matchedProduct.flatMap { productDef =>
88               productDef.tasksList.foldLeft[Result[Accumulator]](Right((tasksForProduct, timeAtStartOfProduct))):
89                 case (accumulatorForTasks, taskId) => accumulatorForTasks.flatMap:
90                   case (taskList, taskStartTime) =>
91                     allTasks.find(_.id == taskId)
92                       .toRight(DomainError.TaskDoesNotExist(taskId.to))
93                       .flatMap { task =>
94                         for {
95                           allocatedPhysicalResources <- allocatePhysicalResources(task.id, task.physicalResources, physicalResources)
96                           allocatedHumanResources <- allocateHumanResources(task.id, task.physicalResources, humanResources)
97                         } yield
98                           val taskEndTime = taskStartTime + task.time.to
99                           val taskSchedule = TaskSchedule(
100                             currentOrder.id,
101                             productInstanceNumber,
102                             task.id,
103                             taskStartTime,
104                             taskEndTime,
105                             allocatedPhysicalResources,
106                             allocatedHumanResources
107                           )
108                           (taskSchedule :: taskList, taskEndTime)
109                       }
110               }
111       }
112     }
113 
114     scheduleResult.map { case (allFinalSchedules, _) =>
115       toXml(allFinalSchedules)
116     }
