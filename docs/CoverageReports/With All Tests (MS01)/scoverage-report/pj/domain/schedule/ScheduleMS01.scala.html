40   // Function to find physical resources available to perform the tasks
41    def allocatePhysicalResources(
42     taskId: TaskId,
43     requiredTypes: List[String],
44     availableResources: List[PhysicalResource]
45   ): Result[List[PhysicalResourceId]] =
46     requiredTypes.foldLeft[Result[(List[PhysicalResourceId], Set[PhysicalResourceId])]](Right((Nil, Set.empty))) {
47       case (accResult, requiredType) => accResult.flatMap { case (assignedIds, usedIds) =>
48         availableResources
49           .find(resource => resource.name.equals(requiredType) && !usedIds.contains(resource.id))
50           .map(_.id)
51           .toRight(DomainError.ResourceUnavailable(taskId.to, requiredType))
52           .map(id => (id :: assignedIds, usedIds + id))
53       }
54     }.map(_._1.reverse)

56   // Function to find human resources available to perform the tasks
57    def allocateHumanResources(
58     taskId: TaskId,
59     requiredTypes: List[String],
60     availableHumans: List[HumanResource]
61   ): Result[List[String]] =
62     requiredTypes.foldLeft[Result[(List[String], Set[String])]](Right((Nil, Set.empty))) {
63       case (accResult, requiredType) => accResult.flatMap { case (assignedNames, usedNames) =>
64         availableHumans
65           .find(hr => hr.physicalResourceTypes.contains(requiredType) && !usedNames.contains(hr.name))
66           .toRight(DomainError.ResourceUnavailable(taskId.to, requiredType))
67           .map(hr => (hr.name :: assignedNames, usedNames + hr.name))
68       }
69     }.map(_._1.reverse)

71   // Function to process all the schedule build logic
72   private def generateSchedule(
73       physicalResources: List[PhysicalResource],
74       physicalTypes: List[String],
75       allTasks: List[Task],
76       humanResources: List[HumanResource],
77       allProducts: List[Product],
78       allOrders: List[Order]
79     ): Result[Elem] =
80 
81     type Accumulator = (List[TaskSchedule], Int)
82     val initialAccumulator: Result[Accumulator] = Right((Nil, 0))
83 
84     val scheduleResult = allOrders.foldLeft[Result[Accumulator]](initialAccumulator) { (accumulatorForOrders, currentOrder) =>
85       accumulatorForOrders.flatMap { case (scheduledTasksSoFar, globalTimeSoFar) =>
86         val matchedProduct = allProducts.find(_.id == currentOrder.productId)
87           .toRight(DomainError.ProductDoesNotExist(currentOrder.productId.to))
88 
89         (1 to currentOrder.quantity.to).foldLeft[Result[Accumulator]](Right((scheduledTasksSoFar, globalTimeSoFar))):
90           case (accumulatorForProducts, productInstanceNumber) => accumulatorForProducts.flatMap:
91             case (tasksForProduct, timeAtStartOfProduct) => matchedProduct.flatMap { productDef =>
92               productDef.tasksList.foldLeft[Result[Accumulator]](Right((tasksForProduct, timeAtStartOfProduct))):
93                 case (accumulatorForTasks, taskId) => accumulatorForTasks.flatMap:
94                   case (taskList, taskStartTime) =>
95                     allTasks.find(_.id == taskId)
96                       .toRight(DomainError.TaskDoesNotExist(taskId.to))
97                       .flatMap { task =>
98                         for {
99                           allocatedPhysicalResources <- allocatePhysicalResources(task.id, task.physicalResourceTypes, physicalResources)
100                           allocatedHumanResources <- allocateHumanResources(task.id, task.physicalResourceTypes, humanResources)
101                           productNumber <- ProductNumber.from(productInstanceNumber)
102                           startTime <- TaskScheduleTime.from(taskStartTime)
103                           endTime <- TaskScheduleTime.from(taskStartTime + task.time.to)
104                         } yield
105                           val taskSchedule = TaskSchedule(
106                             currentOrder.id,
107                             productNumber,
108                             task.id,
109                             startTime,
110                             endTime,
111                             allocatedPhysicalResources,
112                             allocatedHumanResources
113                           )
114                           (taskSchedule :: taskList, endTime.to)
115                       }
116               }
117       }
118     }
119 
120     scheduleResult.map { case (allFinalSchedules, _) =>
121       toXml(allFinalSchedules)
122     }
